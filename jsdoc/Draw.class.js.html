/**
 * The Drawing Handler. Allows to draw on a board.
 * @constructor
 * @param {jQuery} [$board] - The canvas jQuery element.
 */
function Draw(board) {
	this.drawing = 0;
	
	board = board || 0;
	if (board !== 0) this.setBoard(board);
};

/**
 * Sets the board object.
 * @param {Object} board - The board object.
 */
Draw.prototype.setBoard = function(board) {
	this.board = board;
	this.initEvents();
};

/**
 * Initializes the mouse events to the board.
 */
Draw.prototype.initEvents = function() {
	this.board.getBoard().on("mousedown", function(event) {
		// start drawing
		this.drawing = 1;
		this.draw([event.pageX - this.offsetLeft, event.pageY - this.offsetTop]);
		this.redraw();
	}).on("mousemove", function(event) {
		// still drawing
		if (this.drawing == 1) {
			this.draw([event.pageX - this.offsetLeft, event.pageY - this.offsetTop], true);
			this.redraw();
		}
	}).on("mouseup", function(event) {
		// stop drawing
		this.drawing = 0;
	});
};

/**
 * ...
 */
Draw.prototype.draw = function(mouse, drag) {
	drag = drag || false;
	Main.history.add({
		type: HistoryType.DRAW,
		position: mouse,
		drag: drag
	});
};

/**
 * ...
 */
Draw.prototype.redraw = function() {
	this.board.clear();
};
